#include <iostream>
#include "save.h"

class MemoryMonster {
	int* p, _num;
public:
	MemoryMonster(int num) : _num{num} {
		p = new int[num];
		for (int i = 0; i < num; i++) {
			p[i] = num;
		}
	}
	~MemoryMonster() {
		delete[] p;
	}

	void show() {
		std::cout << "Monster-" << _num << std::endl;
	}
};

int main(void)
{
	MemoryMonster mon{123};
	mon.show();
