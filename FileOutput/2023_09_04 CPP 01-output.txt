#include <iostream>
#include <fstream> // 파일을 읽어온다.
#include <chrono>

using namespace std::literals;

int main(void)
{
	//"01.cpp"를 하드디스크에 파일로 저장한다.
	//1. 읽을 파일을 연다.
	std::ifstream ifs {"01.cpp"};			//RAII

	//2. 기록할 파일을 연다.
	std::ofstream ofs{ "2023_09_04 CPP 01-output.txt" };

	
	//3. while(읽을 글자가 있다면 한 글자를 읽는다)
	//		기록할 파일에 쓴다.
	char c;
	ifs >> std::noskipws;
	while (ifs >> c)
		ofs << c;

	ofs.close();
	ofs.open("test.txt", std::ios::out);
	auto now = std::chrono::system_clock::now();
	ofs << now + 9h;


}

#include "save.h"

void save(const std::string& fileName, const std::string& saveName) // 함수의 정의, function definition
{
	std::ifstream ifs{fileName};
	std::ofstream ofs{saveName + ".txt",std::ios::app};

	auto now = std::chrono::system_clock::now(); // 현재 시간을 가져온다.
	auto utc = std::chrono::system_clock::to_time_t(now); // 현재 가져온 시간을 utc로 변환
	auto local_time = std::localtime(&utc); // 지역 시간으로 변환
	//auto old = ofs.imbue(std::locale("ko_KR")); // 출력 스트림 지역 변경

	ofs << '\n' << '\n';
	ofs << "=============================================" << '\n';
	ofs << " Upload Time : " << std::put_time(local_time, "%x %A %X") << '\n';

	ofs << "=============================================" << '\n';
	ofs << '\n';
	// ofs.imbue(old);

	char c;
	ifs >> std::noskipws;
	while (ifs >> c)
		ofs << c;

}
