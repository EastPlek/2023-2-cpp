#include <random>	// 랜덤값과 분포를 사용하려면....

#include "save.h"

std::uniform_int_distribution uid{ 1,6};

int main()
{
	int dice[10000];
	int prob[6];
	for (int i = 0; i < 10000; i++) {
		dice[i] = uid(dre);
		++prob[dice[i] - 1];

	for (int i = 0; i < 6; i++)

//-----------------------------------------------------------------------------
// 내용
// 많은 수의 int가 있다 - 어떻게 저장할건가? 어떤 작업을 하고 싶나?
//-----------------------------------------------------------------------------
// 과제 - 알아서 잘 복습하자
//-----------------------------------------------------------------------------


#include <iostream>
#include <random>	// 랜덤값과 분포를 사용하려면....
#include <format>
#include "save.h"


std::default_random_engine dre; // 랜덤값을 생성하는 엔진
std::uniform_int_distribution uid{ 1,6};

int main()
{
	//숫자와 변수를 비교할 때는 숫자를 왼편에 쓴다.
	int prob[6]{};
	for (int i = 0; i < 1'0000; i++) {
		prob[uid(dre) - 1]++;
	}

	for (int i = 0; i < 6; i++)
	{
		std::cout << i + 1 << " - " << prob[i] << "번, 확률 -  " << std::format("{:2.1f}", prob[i] / static_cast < float>(100)) << "%" << '\n';
