#include "save.h"

// [문제] 사용자가 정수 num을 하나 입력한다.
//int를 num개 만큼 저장할 공간을 확보한다.
// 그 공간을 1 부터 시작하는 정수로 채운다. [1,num]
// int값의 합계를 화면에 출력하라.
// 이 과정을 계속 반복하라.
int main() // function call operator
{
	save("17.cpp", "FileOutput\\2023_10_09 CPP 17output", std::ios::app, true);

	while (true) 
	{
		int num{};
		int* p{};
		std::cout << "정수 하나를 입역해주세요 : ";
		std::cin >> num;
		p = new int[num];
		for (int i = 0; i < num; i++) {
			p[i] = i + 1;
		}
		//합계 계산
		int sum{0};
		for (int i = 0; i < num; i++) {
			sum += p[i];
		}
		std::cout << "합계 : " << sum << "\n\n";
		delete[] p;
	}
}
