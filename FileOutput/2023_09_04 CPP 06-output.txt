#include <iostream>
#include <random>	// 랜덤값과 분포를 사용하려면....
#include <format>
#include "save.h"


std::default_random_engine dre; // 랜덤값을 생성하는 엔진
std::uniform_int_distribution uid{ 1,999 };

int main()
{
	int numbers[100];
	for (int i = 0; i < 100; i++)
		numbers[i] = uid(dre);

	for (int num : numbers)
		std::cout << std::format("{:4d}", num);


	save("06.cpp", "FileOutput\\2023_09_04 CPP 06-output", std::ios::app, true);
}
